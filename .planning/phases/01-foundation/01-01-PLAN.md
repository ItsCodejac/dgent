  <name>Task 1: Initialize npm package with TypeScript</name>
  <files>package.json, tsconfig.json, .gitignore</files>
  <action>
    Run `npm init -y` and configure package.json:
    - name: "dgent"
    - description: "Strip AI tells from agent output"
    - bin: { "dgent": "./dist/cli.js" }
    - main: "./dist/index.js"
    - scripts: { "build": "tsc", "dev": "tsc --watch" }
    - type: "module" (ESM)
    - engines: { "node": ">=18" }
    - license: "MIT"

    Install dev dependencies: typescript, @types/node

    Create tsconfig.json:
    - target: "ES2022"
    - module: "Node16"
    - moduleResolution: "Node16"
    - outDir: "./dist"
    - rootDir: "./src"
    - strict: true
    - esModuleInterop: true
    - declaration: true

    Create .gitignore: node_modules/, dist/, .DS_Store
  </action>
  <verify>npm install succeeds, npx tsc --noEmit exits 0 (no source files yet is fine)</verify>
  <done>package.json has correct bin/scripts/type fields, tsconfig.json compiles to dist/, .gitignore excludes node_modules and dist</done>
</task>
